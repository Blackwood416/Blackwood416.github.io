Create `scripts/migrate-hexo.mjs` with:

```js
import fs from 'node:fs/promises';
import path from 'node:path';
import process from 'node:process';
import matter from 'gray-matter';

const DEFAULT_HEXO_ROOT = 'F:/hexo-blogs';

export function normalizeList(value) {
  if (value === undefined || value === null) return [];
  const values = Array.isArray(value) ? value : [value];
  return values.map((item) => String(item).trim()).filter(Boolean);
}

function stripMarkdownInline(markdown) {
  return markdown
    .replace(/!\[[^\]]*]\([^)]*\)/g, '')
    .replace(/\[([^\]]+)]\([^)]*\)/g, '$1')
    .replace(/[`*_~>#-]/g, '')
    .replace(/<[^>]+>/g, '')
    .replace(/\s+/g, ' ')
    .trim();
}

export function extractDescription(markdown) {
  const withoutCode = markdown.replace(/```[\s\S]*?```/g, '');
  const paragraphs = withoutCode.split(/\n{2,}/);

  for (const paragraph of paragraphs) {
    const trimmed = paragraph.trim();
    if (!trimmed) continue;
    if (trimmed.startsWith('#')) continue;
    if (trimmed.startsWith('|')) continue;
    if (/^!\[[^\]]*]\([^)]*\)/.test(trimmed)) continue;

    const clean = stripMarkdownInline(trimmed);
    if (clean.length > 0) {
      return clean.length > 140 ? `${clean.slice(0, 137)}...` : clean;
    }
  }

  return 'Blackwood 的技术博客文章。';
}

export function removeDuplicateTitleHeading(markdown, title) {
  const lines = markdown.replace(/^\uFEFF/, '').split(/\r?\n/);
  let index = 0;
  while (index < lines.length && lines[index].trim() === '') index += 1;

  if (index < lines.length && lines[index].trim() === `# ${title}`) {
    lines.splice(index, 1);
    while (index < lines.length && lines[index].trim() === '') {
      lines.splice(index, 1);
    }
    return lines.join('\n').trimStart();
  }

  return markdown;
}

export function findRelativeImageReferences(markdown) {
  const matches = [];
  const imagePattern = /!\[[^\]]*]\(([^)\s]+)(?:\s+"[^"]*")?\)/g;
  for (const match of markdown.matchAll(imagePattern)) {
    const target = match[1].trim();
    if (/^(https?:|data:|\/|#)/i.test(target)) continue;
    matches.push(target);
  }
  return matches;
}

async function pathExists(filePath) {
  try {
    await fs.access(filePath);
    return true;
  } catch {
    return false;
  }
}

function sortObjectKeys(data) {
  return {
    title: data.title,
    description: data.description,
    pubDate: data.pubDate,
    tags: data.tags,
    categories: data.categories,
    draft: data.draft,
  };
}

async function emptyMarkdownDirectory(directory) {
  await fs.mkdir(directory, { recursive: true });
  const entries = await fs.readdir(directory, { withFileTypes: true });
  await Promise.all(
    entries
      .filter((entry) => entry.isFile() && /\.(md|mdx)$/i.test(entry.name))
      .map((entry) => fs.rm(path.join(directory, entry.name))),
  );
}

async function migratePost({ inputPath, outputDirectory }) {
  const source = await fs.readFile(inputPath, 'utf8');
  const parsed = matter(source);
  const title = String(parsed.data.title ?? path.basename(inputPath, '.md')).trim();
  const body = removeDuplicateTitleHeading(parsed.content, title);
  const relativeImages = findRelativeImageReferences(body);
  const outputPath = path.join(outputDirectory, path.basename(inputPath));

  const frontmatter = sortObjectKeys({
    title,
    description: extractDescription(body),
    pubDate: String(parsed.data.date ?? '').trim(),
    tags: normalizeList(parsed.data.tags),
    categories: normalizeList(parsed.data.categories),
    draft: false,
  });

  const output = matter.stringify(`${body.trim()}\n`, frontmatter, {
    lineWidth: 100,
  });

  await fs.writeFile(outputPath, output, 'utf8');
  return { outputPath, relativeImages };
}

async function migrateAbout({ inputPath, outputPath }) {
  const source = await fs.readFile(inputPath, 'utf8');
  const parsed = matter(source);
  const title = String(parsed.data.title ?? '关于').trim();
  const output = matter.stringify(`${parsed.content.trim()}\n`, { title }, { lineWidth: 100 });
  await fs.mkdir(path.dirname(outputPath), { recursive: true });
  await fs.writeFile(outputPath, output, 'utf8');
}

export async function migrateHexo(options = {}) {
  const cwd = options.cwd ?? process.cwd();
  const hexoRoot = options.hexoRoot ?? process.env.HEXO_ROOT ?? DEFAULT_HEXO_ROOT;
  const postsDirectory = path.join(hexoRoot, 'source', '_posts');
  const aboutPath = path.join(hexoRoot, 'source', 'about', 'index.md');
  const outputPostsDirectory = path.join(cwd, 'src', 'content', 'blog');
  const outputAboutPath = path.join(cwd, 'src', 'content', 'pages', 'about.md');

  await emptyMarkdownDirectory(outputPostsDirectory);

  const entries = await fs.readdir(postsDirectory, { withFileTypes: true });
  const postFiles = entries
    .filter((entry) => entry.isFile() && entry.name.endsWith('.md'))
    .map((entry) => path.join(postsDirectory, entry.name))
    .sort((a, b) => a.localeCompare(b));

  const relativeImageFindings = [];
  for (const inputPath of postFiles) {
    const result = await migratePost({ inputPath, outputDirectory: outputPostsDirectory });
    for (const imagePath of result.relativeImages) {
      const resolved = path.join(path.dirname(inputPath), imagePath);
      relativeImageFindings.push({
        post: path.basename(inputPath),
        imagePath,
        exists: await pathExists(resolved),
      });
    }
  }

  await migrateAbout({ inputPath: aboutPath, outputPath: outputAboutPath });

  return {
    posts: postFiles.length,
    about: outputAboutPath,
    relativeImages: relativeImageFindings,
  };
}

if (import.meta.url === `file://${process.argv[1].replace(/\\/g, '/')}`) {
  const result = await migrateHexo();
  console.log(`Migrated ${result.posts} posts.`);
  console.log(`Migrated About page to ${result.about}.`);

  const missing = result.relativeImages.filter((item) => !item.exists);
  if (result.relativeImages.length > 0) {
    console.log('Relative image references:');
    for (const item of result.relativeImages) {
      const status = item.exists ? 'found' : 'missing';
      console.log(`- ${item.post}: ${item.imagePath} (${status})`);
    }
  }

  if (missing.length > 0) {
    process.exitCode = 0;
  }
}
```

function routeSegment(value) {
  return encodeURIComponent(value);

function countValues(posts, key) {
  const counts = new Map();
  for (const post of posts) {
    for (const value of post.data[key] ?? []) {
      counts.set(value, (counts.get(value) ?? 0) + 1);
    }
  }
  return [...counts.entries()]
    .map(([name, count]) => ({ name, count }))
    .sort((a, b) => b.count - a.count || a.name.localeCompare(b.name, 'zh-CN'));
}

function groupByYearMonth(posts) {
  const groups = new Map();
  for (const post of posts) {
    const date = new Date(post.data.pubDate);
    const year = String(date.getFullYear());
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const key = `${year}-${month}`;
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key).push(post);
  }
  return [...groups.entries()].map(([key, posts]) => ({ key, posts }));
}

```ts
import type { CollectionEntry } from 'astro:content';

export type BlogPost = CollectionEntry<'blog'>;
export type TaxonomyCount = { name: string; count: number };
export type ArchiveGroup = { key: string; year: string; month: string; posts: BlogPost[] };

export function getVisiblePosts(posts: BlogPost[]) {
  return posts
    .filter((post) => !post.data.draft)
    .sort((a, b) => b.data.pubDate.valueOf() - a.data.pubDate.valueOf());
}

export function routeSegment(value: string) {
  return encodeURIComponent(value);
}

export function countTaxonomy(posts: BlogPost[], key: 'tags' | 'categories'): TaxonomyCount[] {
  const counts = new Map<string, number>();

  for (const post of posts) {
    for (const value of post.data[key] ?? []) {
      counts.set(value, (counts.get(value) ?? 0) + 1);
    }
  }

  return [...counts.entries()]
    .map(([name, count]) => ({ name, count }))
    .sort((a, b) => b.count - a.count || a.name.localeCompare(b.name, 'zh-CN'));
}

export function groupPostsByYearMonth(posts: BlogPost[]): ArchiveGroup[] {
  const groups = new Map<string, ArchiveGroup>();

  for (const post of posts) {
    const year = String(post.data.pubDate.getFullYear());
    const month = String(post.data.pubDate.getMonth() + 1).padStart(2, '0');
    const key = `${year}-${month}`;

    if (!groups.has(key)) {
      groups.set(key, { key, year, month, posts: [] });
    }

    groups.get(key)?.posts.push(post);
  }

  return [...groups.values()].sort((a, b) => b.key.localeCompare(a.key));
}
```
